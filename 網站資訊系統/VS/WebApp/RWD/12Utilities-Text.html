Ajax and WebAPI Code Restful

        <script>
        
                var item = "";
                $.ajax({
                    type: 'get',
                    url: 'https://api.kcg.gov.tw/api/service/get/a0d61f1f-5727-477d-bde0-218f88a46813',
                    success: function (data) {
                        console.log(data.data);

                        item = "<table border='1' width=1200 style='margin:auto'>";
                        item += "<tr>";
                        item += "<th>年度</th>";
                        item += "<th>項目</th>";
                        item += "<th>稅目別</th>";

                        item += "</tr>";
                        for (i = 0; i < data.data.length; i++) {
                            item += "<tr>";
                            item += "<td>" + data.data[i].資料年度 + "</td>";
                            item += "<td>" + data.data[i].統計項目 + "</td>";
                            item += "<td>" + data.data[i].稅目別 + "</td>";

                            item += "</tr>";
                        }
                        item += "</table>";

                        $('#container').append(item);
                    },
                    error: function () {
                        alert('Error!!');
                    }
                });
            
        </script>
